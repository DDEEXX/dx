В этом каталоге файлы с параметрами виджетов в формате JSON
{
    "label" : имя датчика в БД
    "image" : иконка виджета
    "graph" : { параметры графика
        "type" : тип (линейный, столбчатый)
        "variant" : вариант (минимальный, нормальный)
        "width" : ширина
        "height" : высота
    }
}
